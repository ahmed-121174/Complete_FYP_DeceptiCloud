\begin{frame}{Project Description}
  \textbf{\textcolor{dcCyan}{What is DeceptiCloud?}}\\[0.3em]
  An AI-powered cyber deception platform that protects cloud-hosted web applications
  by intelligently detecting cyberattacks and luring attackers into convincing fake
  environments (honeypots), while keeping real systems completely secure.

  \vspace{0.5em}
  \textbf{\textcolor{dcCyan}{The Core Idea:}}
  \begin{itemize}
    \item Every incoming HTTP request passes through an AI gateway
    \item Machine Learning classifies it as \textcolor{dcRed}{malicious} or \textcolor{dcGreen}{benign} in real time
    \item Malicious requests are silently rerouted to \textcolor{dcYellow}{honeypots}
    \item Attacker interacts with a fake system --- revealing their techniques
    \item All activity logged to a \textcolor{dcPurple}{tamper-proof blockchain}
  \end{itemize}

  \vspace{0.4em}
  \begin{block}{}
    \centering\textit{``We deceive the attacker before they can deceive us.''}
  \end{block}
\end{frame}

\begin{frame}{Problem Statement}
  \textbf{\textcolor{dcRed}{Why does this project matter?}}
  \begin{itemize}
    \item Cyberattacks cost the global economy \textbf{\$8 trillion in 2023} (Cybersecurity Ventures)
    \item Traditional security only \textbf{blocks} attacks --- it does not \textbf{study} them
    \item Attackers probe systems repeatedly, improving techniques each run
    \item No existing open-source system combines:\\[0.2em]

          \texttt{+} \textcolor{dcGreen}{LLM deception}
          \texttt{+} \textcolor{dcPurple}{blockchain evidence} in one platform

  \end{itemize}

  \vspace{0.5em}
  \begin{block}{The Gap}
    Existing tools \textbf{detect} attacks.\\
    DeceptiCloud \textbf{detects, deceives, documents, and learns}.
  \end{block}

\begin{frame}{Dynamic Honeypot System --- Core Module}
  \textbf{\textcolor{dcCyan}{What is a Honeypot?}}
  A decoy system that attracts attackers, appears vulnerable, and records everything
  without exposing any real data or real users.

  \vspace{0.3em}
  \centering\footnotesize
  \begin{tabular}{@{} p{2.3cm} p{3.0cm} p{5.0cm} @{}}
    \toprule
    \rowcolor{dcMid}
    \textcolor{dcCyan}{Dimension} &
    \textcolor{dcRed}{Static Honeypot} &
    \textcolor{dcGreen}{DeceptiCloud Dynamic}\\
    \midrule
    Content       & Fixed HTML, never changes     & LLM generates unique content per session\\
    \rowcolor{dcDark}
    Database      & Empty / manual fake data      & GAN: 35 realistic fake users per site\\
    Routing       & All traffic hits blindly      & ML-only routes confirmed attackers\\
    \rowcolor{dcDark}
    Detectability & Fingerprintable               & New LLM output each session\\
    Monitoring    & Basic logs                    & Blockchain + canary + SOC dashboard\\
    \rowcolor{dcDark}
    Recovery      & No self-healing               & K8s watchdog restarts in $\sim$15s\\
    \bottomrule
  \end{tabular}

  \vspace{0.4em}
  \begin{block}{}
    \centering\small\textit{``A static honeypot looks like a trap.
    A dynamic honeypot looks like the real thing.''}
  \end{block}
\end{frame}

\begin{frame}[fragile]{Module 1: AI Routing Proxy (Port 8080)}
  \textbf{\textcolor{dcCyan}{Purpose:}} Central intelligence gateway --- every HTTP request passes through it.

  \vspace{0.3em}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{What it does:}
      \begin{itemize}\small
        \item Extracts 15+ features from every request
        \item Rule-based engine (regex) + ML model
        \item \textcolor{dcRed}{Malicious} $\rightarrow$ Honeypot (4001--4007)
        \item \textcolor{dcGreen}{Benign} $\rightarrow$ Real site (3001--3007)
        \item Attacker never knows they were rerouted
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
\begin{lstlisting}
Request
 -> Feature Extraction
 -> Rule Score (regex)
 -> ML Model Score
 -> Score >= 0.5?
    YES -> Honeypot
           -> LLM Response
    NO  -> Real Site
           -> Normal response
\end{lstlisting}
    \end{column}
  \end{columns}

  \vspace{0.3em}
  \begin{block}{Key Achievement}
    \textbf{100\% detection rate} on all 16 tested attack types.
    Zero impact on legitimate users.
  \end{block}
\end{frame}

\begin{frame}{Module 3: LLM Response Engine}

      \textbf{\textcolor{dcGreen}{Why LLM, not static pages?}}
      \begin{itemize}\small
        \item Static pages are fingerprintable
        \item LLM generates unique, contextual HTML per session
        \item Attacker believes they compromised the system
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{\textcolor{dcYellow}{Technical:}}
      \begin{itemize}\small
        \item Runs entirely on-device (no cloud API)
        \item Cached within session for speed
        \item Cache cleared on restart for variety
      \end{itemize}

      \begin{itemize}

      \textbf{\textcolor{dcCyan}{Why GAN?}}
      \begin{itemize}
        \item Empty DBs signal ``honeypot'' to smart attackers

        \item GAN output is statistically indistinguishable
      \end{itemize}

        \item Invisible to attacker
        \item Identifies leaked data later
      \end{itemize}
      \vspace{0.4em}
      \begin{block}{Practical Value}
        \small Attacker who extracts the honeypot DB gets
        \textbf{100\% fake data}. Real users completely protected.
      \end{block}
